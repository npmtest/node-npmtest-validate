function each (fn) {
  return function (arr) {
    if (arr == null) return true;
    if (!Array.isArray(arr)) return false;
 
    for (var i = 0; i < arr.length; i++) {
      if (!fn(arr[i])) return false;
    }
 
    return true;
  }
}
